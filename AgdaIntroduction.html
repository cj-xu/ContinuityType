{-# OPTIONS --without-K #-}

module AgdaIntroduction where

data ℕ : Set where
 zero : ℕ
 succ : ℕ → ℕ

data ₂ : Set where
 ₀ : ₂
 ₁ : ₂

₂ℕ : Set
₂ℕ = ℕ → ₂

data Vec (A : Set) : ℕ → Set where
 ⟨⟩ : Vec A zero
 _∷_ : {n : ℕ} → A → Vec A n → Vec A (succ n)

Π : {A : Set} → (A → Set) → Set
Π {A} B = (a : A) → B a

data Σ {A : Set} (B : A → Set) : Set where
 _,_ : (a : A) → (b : B a) → Σ B

pr₁ : {A : Set} {B : A → Set} → (Σ \(a : A) → B a) → A
pr₁ (a , _) = a

pr₂ : {A : Set} {B : A → Set} → (w : Σ \(a : A) → B a) → B(pr₁ w)
pr₂ (_ , b) = b

_×_ : Set → Set → Set
A × B = Σ \(a : A) → B

data _≡_ {A : Set} : A → A → Set where
 refl : {a : A} → a ≡ a

J : {A : Set} {C : (x y : A) → x ≡ y → Set}
  → ((a : A) → C a a refl)
  → (x y : A) → (p : x ≡ y) → C x y p
J d a .a refl = d a

_≅_ : Set → Set → Set
A ≅ B = Σ \(f : A → B) → Σ \(g : B → A) →
          (∀(a : A) → g(f a) ≡ a) × (∀(b : B) → f(g b) ≡ b)
